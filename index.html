Class

Basic

console.clear();
class Point {
    constructor() {
        this.setTo(...arguments)
    }
    setTo(x = 0, y = x) {
        this.x = x;
        this.y = y;
        return this;
    }
    static create(){
        return new Point(...arguments);
    }
}
var p = new Point().setTo(1);
var q = Point.create(3, 4);
console.log(p, q);

Inheritance

class Vector extends Point {
    constructor() {
        super(...arguments);
    }
    setTo(position = 0){
        if (position.x) {
            let {x, y} = position;
            return super.setTo(x, y);
        } else if (position.length) {
            return super.setTo(...position);
        } else return super.setTo(...arguments);
    }

}
var v = new Vector().setTo([3, 4]);
console.log(v);

Getter & Setter

class Vector extends Point {
    //...
    normalize() {
        var m = this.magnitude;
        this.x /= m;
        this.y /= m;
        return this;
    }
    get magnitude() {
        var {x, y} = this;
        return Math.sqrt(x*x + y*y) || 1;
    }
    set magnitude(magnitude) {
        this.normalize();
        this.x *= magnitude;
        this.y *= magnitude;
    }
}
var v = new Vector().setTo([3, 4]);
    v.magnitude = 10;
console.log(v);
